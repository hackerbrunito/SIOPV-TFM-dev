#!/usr/bin/env bash
# Pre-write hook: Anti-autopilot checkpoint reminder
# Event: PreToolUse (Write, Edit)
# Output format: hookSpecificOutput with permissionDecision

TODAY=$(date +%Y%m%d)
SIOPV_ROOT="/Users/bruno/siopv"
MISSING_DOCS=()

if [[ ! -f "${SIOPV_ROOT}/.build/checkpoints/daily/briefing-read-${TODAY}" ]]; then

    cat <<EOF
{
  "hookSpecificOutput": {
    "hookEventName": "PreToolUse",
    "permissionDecision": "ask",
    "permissionDecisionReason": "Faltan checkpoints de lectura de documentacion SIOPV.",
    "additionalContext": "RECORDATORIO: Antes de Write/Edit, verifica que hayas consultado: ${DOCS_LIST}. Esto previene errores de piloto automatico."
  }
}
EOF
    exit 0
fi

cat <<EOF
{
  "hookSpecificOutput": {
    "hookEventName": "PreToolUse",
    "permissionDecision": "allow"
  }
}
EOF
exit 0

```bash
#!/usr/bin/env bash
# Framework self-testing script — SIOPV
# Validates SIOPV .claude/ configuration integrity
# Run manually: bash .claude/hooks/test-framework.sh
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/../.." && pwd)"
CLAUDE_DIR="$PROJECT_ROOT/.claude"

PASS=0
FAIL=0

echo "=== SIOPV Framework Validation ==="

pass() {
  echo "[PASS] $1"
  PASS=$((PASS + 1))
}

fail() {
  echo "[FAIL] $1"
  FAIL=$((FAIL + 1))
}

# ============================================================
# 1. settings.json structure
# ============================================================

SETTINGS="$CLAUDE_DIR/settings.json"

if [ ! -f "$SETTINGS" ]; then
  fail "settings.json: file not found"
else
  if jq -e '.permissions.deny | type == "array"' "$SETTINGS" > /dev/null 2>&1; then
    pass "settings.json: permissions.deny exists (array)"
  else
    fail "settings.json: missing or invalid permissions.deny"
  fi

  if jq -e '.permissions.ask | type == "array"' "$SETTINGS" > /dev/null 2>&1; then
    pass "settings.json: permissions.ask exists (array)"
  else
    fail "settings.json: missing or invalid permissions.ask"
  fi

  if jq -e '.permissions.allow | type == "array"' "$SETTINGS" > /dev/null 2>&1; then
    pass "settings.json: permissions.allow exists (array)"
  else
    fail "settings.json: missing or invalid permissions.allow"
  fi
fi

# ============================================================
# 2. Agent YAML frontmatter validation
# ============================================================

AGENTS_DIR="$CLAUDE_DIR/agents"
VALID_MODELS="sonnet haiku opus"
REQUIRED_AGENT_FIELDS="name description tools model"

for agent_file in "$AGENTS_DIR"/*.md; do
  [ -f "$agent_file" ] || continue
  basename_file=$(basename "$agent_file")
  frontmatter=$(sed -n '/^---$/,/^---$/p' "$agent_file" | sed '1d;$d')

  if [ -z "$frontmatter" ]; then
    fail "agent $basename_file: no YAML frontmatter found"
    continue
  fi

  for field in $REQUIRED_AGENT_FIELDS; do
    value=$(echo "$frontmatter" | grep "^${field}:" | head -1 | sed "s/^${field}:[[:space:]]*//" | sed 's/^"//' | sed 's/"$//')
    if [ -z "$value" ]; then
      fail "agent $basename_file: missing '$field' field"
    else
      case "$field" in
        model)
          if echo "$VALID_MODELS" | grep -qw "$value"; then
            pass "agent $basename_file: model '$value' is valid"
          else
            fail "agent $basename_file: model '$value' is not one of: $VALID_MODELS"
          fi
          ;;
        *)
          pass "agent $basename_file: has '$field' field"
          ;;
      esac
    fi
  done
done

# ============================================================
# 3. Workflow files validation
# ============================================================

WORKFLOW_DIR="$CLAUDE_DIR/workflow"
WORKFLOW_FILES=(
  "briefing.md"
  "compaction-log.md"
)

for wf in "${WORKFLOW_FILES[@]}"; do
  wf_path="$WORKFLOW_DIR/$wf"
  if [ ! -f "$wf_path" ]; then
    fail "workflow $wf: file not found"
  else
    pass "workflow $wf: exists"
  fi
done

# ============================================================
# Results
# ============================================================

echo ""
echo "=== Results: $PASS passed, $FAIL failed ==="

if [ "$FAIL" -gt 0 ]; then
  exit 1
fi

exit 0
```

- `/Users/bruno/siopv/.claude/workflow/briefing.md`
- `/Users/bruno/siopv/.claude/workflow/compaction-log.md`

**SIOPV version — full script content:**
```bash
#!/usr/bin/env bash
# pre-compact.sh — Runs before every Claude Code compaction event (SIOPV)
#
# Purpose:
#   1. Update the "Last updated" timestamp in briefing.md
#   2. Append a one-line entry to compaction-log.md
#
# Safety: Never blocks compaction — always exits 0.

trap 'exit 0' ERR

BRIEFING="/Users/bruno/siopv/.claude/workflow/briefing.md"
COMPACT_LOG="/Users/bruno/siopv/.claude/workflow/compaction-log.md"
TIMESTAMP="$(date -u +%Y-%m-%dT%H:%M:%SZ)"

if [[ ! -f "$BRIEFING" ]]; then
  echo "[pre-compact WARNING] briefing.md not found at $BRIEFING — skipping update" >&2
  exit 0
fi

if grep -q "^> Last updated:" "$BRIEFING"; then
  perl -i -pe "s|^> Last updated:.*|> Last updated: ${TIMESTAMP}|" "$BRIEFING"
else
  echo "[pre-compact WARNING] Could not find '> Last updated:' marker in briefing.md" >&2
fi

if [[ ! -f "$COMPACT_LOG" ]]; then
  cat > "$COMPACT_LOG" <<'EOF'
# Compaction Log

Entries added automatically by pre-compact.sh before each compaction event.

EOF
fi

echo "- ${TIMESTAMP} — PreCompact fired (manual/auto)" >> "$COMPACT_LOG"

exit 0
```

**Cross-file dependencies:**
- `/Users/bruno/siopv/.claude/workflow/briefing.md` (must contain `> Last updated:` marker line)
- `/Users/bruno/siopv/.claude/workflow/compaction-log.md` (created on first run if absent)
